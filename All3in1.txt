Table marcas {
  id int [pk, increment]
  nombre varchar(50) [unique, not null]
  activo boolean [default: true]
  fecha_creacion timestamp [default: `CURRENT_TIMESTAMP`]
  // Nota: Catálogo de marcas de vehículos (Toyota, Honda, etc.)
}

Table modelos {
  id int [pk, increment]
  marca_id int [not null]
  nombre varchar(100) [not null]
  activo boolean [default: true]
  fecha_creacion timestamp [default: `CURRENT_TIMESTAMP`]
  // Nota: Modelos específicos por marca (Corolla, Civic, etc.)
}

Table generaciones {
  id int [pk, increment]
  modelo_id int [not null]
  nombre varchar(50) [not null]
  descripcion varchar(200)
  anio_inicio int [not null]
  anio_fin int [not null]
  total_investment decimal(15,2) [default: 0.00]
  total_ingresos decimal(15,2) [default: 0.00]
  total_egresos decimal(15,2) [default: 0.00]
  balance_neto decimal(15,2) [default: 0.00]
  activo boolean [default: true]
  fecha_creacion timestamp [default: `CURRENT_TIMESTAMP`]
  // Nota: Generaciones por modelo con control financiero completo
  // Los campos totales se actualizan automáticamente via triggers
}

Table vehiculos {
  id int [pk, increment]
  codigo_vehiculo varchar(20) [unique]
  generacion_id int [not null]
  imagen_url text
  anio int [not null]
  precio_compra decimal(12,2) [not null]
  costo_grua decimal(10,2) [default: 0]
  comisiones decimal(10,2) [default: 0]
  inversion_total decimal(12,2) [note: "Calculado: precio_compra + costo_grua + comisiones"]
  costo_recuperado decimal(12,2) [default: 0, note: "Recuperación por ventas de partes"]
  costo_pendiente decimal(12,2) [note: "Calculado: inversion_total - costo_recuperado"]
  fecha_ingreso date [not null]
  estado enum('DISPONIBLE', 'VENDIDO', 'DESARMADO', 'REPARACION') [default: 'DISPONIBLE']
  precio_venta decimal(12,2)
  fecha_venta date
  activo boolean [default: true]
  notas text
  fecha_creacion timestamp [default: `CURRENT_TIMESTAMP`]
  fecha_actualizacion timestamp [default: `CURRENT_TIMESTAMP`, note: "ON UPDATE CURRENT_TIMESTAMP"]
  // Nota: Vehículos individuales con seguimiento financiero completo
  // El estado cambia automáticamente al registrar ventas
}

PROCESOS AUTOMATIZADOS:

1. COMPRA DE VEHÍCULO:
   - INSERT en vehiculos → trigger genera código
   - trigger tr_insertar_compra_vehiculo crea transacción financiera
   - trigger tr_actualizar_totales_generacion_insert actualiza totales

2. DESARME DE VEHÍCULO:
   - INSERT en inventario_repuestos → trigger genera código
   - trigger tr_generar_codigo_ubicacion genera ubicación
   - trigger tr_actualizar_totales_vehiculo_insert actualiza inversión

3. VENTA DE VEHÍCULO:
   - INSERT en transacciones_financieras (Venta Vehículo)
   - trigger tr_venta_vehiculo_actualiza_estado cambia estado a VENDIDO
   - trigger tr_actualizar_costo_recuperado_vehiculo actualiza recuperación
   - trigger tr_actualizar_totales_generacion_insert actualiza totales

4. VENTA DE REPUESTO:
   - INSERT en transacciones_financieras (Venta Repuesto)
   - trigger tr_actualizar_estado_repuesto_por_venta cambia estado
   - trigger tr_actualizar_costo_recuperado_vehiculo actualiza recuperación vehículo origen
   - trigger tr_actualizar_totales_generacion_insert actualiza totales

5. COMPRA DE REPUESTO SIN VEHÍCULO:
   - CALL sp_insertar_repuesto_con_generacion_sin_vehiculo
   - Procedimiento inserta repuesto y crea transacción de compra
   - Triggers actualizan totales automáticamente
